.loader {
  position: fixed;
  left: 0;
  top: 0;
  z-index: 100000;
  width: 100%;
  height: 100%;
  background: rgba(255, 255, 255, 0.5);
  display: flex;
  flex-direction: column;
  justify-content: center;
  align-items: center;

  .spin {
    width: 8rem;
    height: 8rem;
    border-radius: 50%;
    border: 0.8rem solid #ccc;
    border-top-color: #111;
    animation: loader 1s linear infinite;
  }

  .id_msg {
    margin-top: 2rem;
    font-size: 1.6rem;
    color: #333;
    font-weight: 500;
  }
}

@keyframes loader {
  0% {
    transform: rotate(0deg);
  }
  100% {
    transform: rotate(360deg);
  }
}
function fn_action_printing() {
      loader("on", "잠시만 기다려주세요.");
      setTimeout(function () {
        window.open("print.html", "", "width=1280,height=900,left=0,scrollbars=yes");
        loader("off");
      }, 2000);
      return false;
    }

    function loader(mode, msg) {
      if (mode == "on") {
        $("body").append('
' + msg + "
");
      } else {
        $(".loader").remove();
      }
    }
